```go
// GetTransformer returns the appropriate transformer based on the bank
func (f *TransformerFactory) GetTransformer(bank string) (mozart.MozartTransformer, errors.IError) {
	switch bank {
	case "rbl":
		return rbl.NewRBLTransformer(), nil
	case "icici":
		return icici.NewICICITransformer(), nil
	case "axis":
		return axis.NewAxisTransformer(), nil
	case "yesbank":
		return yesbank.NewYesBankTransformer(), nil
	case "idfc":
		return idfc.NewIDFCTransformer(), nil
	case "BANKNAME":  // ADD THIS CASE - use lowercase bank name
		return BANKNAME.NewBANKNAMETransformer(), nil  // use package name and constructor

	default:
		return nil, errors.New("unsupported bank type")
	}
}
```

INTEGRATION RULES:
1. Bank name string must be lowercase (e.g., "sbi", "hdfc", "kotak")
2. Package import path must match: internal/transformer/mozart/BANKNAME
3. Constructor function must follow pattern: NewBANKNAMETransformer()
4. Constructor must return pointer to transformer struct
5. Bank name must match channel enum string representation

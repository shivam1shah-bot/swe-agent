// BankNameTransformer implements the Transformer interface for Bank BANKNAME
type BankNameTransformer struct {}

// NewBankNameTransformer returns a new instance of BankNameTransformer
func NewBankNameTransformer() *BankNameTransformer {
	return &BankNameTransformer{}
}

// TransformToRequest transforms the request to Bank's Mozart request format
func (t *BankNameTransformer) TransformToRequest(
	accountID string,
	merchantID string, 
	cred banking_accounts.FetchBankingCredsResponse) (
	mozart2.Request,
	*mozart2.RequestAttributes,
	errors.IError) {
	
	if accountID == "" {
		return mozart2.Request{}, nil, errtype.InternalServerError.
			New(errcode.GatewayBadRequest)
	}
	
	var credentials models.Credentials
	if credentialsData, exists := cred.Data["credentials"]; exists {
		credentialsJSON, err := json.Marshal(credentialsData)
		if err != nil {
			return mozart2.Request{}, nil, errtype.InternalServerError.New(errcode.MarshalError)
		}
		err = json.Unmarshal(credentialsJSON, &credentials)
		if err != nil {
			return mozart2.Request{}, nil, errtype.InternalServerError.New(errcode.UnmarshalError)
		}
	} else {
		return mozart2.Request{}, nil, errtype.InternalServerError.New(errcode.UnmarshalError)
	}
	
	mozartRequest := models.MozartBalanceFetchRequest{
		SourceAccount: models.SourceAccount{
			AccountNumber: accountID,
			Credentials:   credentials,
		},
	}
	
	requestAttributes := getMozartAttributes()
	err := ValidateRequest(mozartRequest)
	if err != nil {
		return mozart2.Request{}, nil, err.
			WithInternalMetadata(map[string]string{
				"merchant_id": merchantID,
			})
	}
	
	return mozart2.Request{Entities: mozartRequest}, &requestAttributes, nil
}

// TransformToResponse transforms Mozart's response into common fetch service response
func (t *BankNameTransformer) TransformToResponse(response mozart2.Response) (
	*mozart2.BalanceFetchResponse,
	errors.IError) {
	
	jsonData, err := json.Marshal(response.Data)
	if err != nil {
		return nil, errtype.InternalServerError.
			New(errcode.MarshalError).
			Wrap(err)
	}

	var mozartResponse models.Data
	err = json.Unmarshal(jsonData, &mozartResponse)
	if err != nil {
		return nil, errtype.InternalServerError.
			New(errcode.UnmarshalError).
			Wrap(err)
	}

	floatValue, err := strconv.ParseFloat(mozartResponse.AccountBalanceAmount, 64)
	if err != nil {
		return nil, errtype.InternalServerError.
			New(errcode.ParseFloatError).
			Wrap(err)
	}

	balanceInt, err := intutils.FormatAmount(floatValue)
	if err != nil {
		return nil, errtype.InternalServerError.
			New(errcode.AmountConversionError).
			Wrap(err)
	}

	return &mozart2.BalanceFetchResponse{
		Amount: int64(balanceInt),
	}, nil
}

func getMozartAttributes() mozart2.RequestAttributes {
	return mozart2.RequestAttributes{
		Namespace:                        mozart2.FTSNamespace,
		Gateway:                          "bankname",  // Use lowercase bank name
		Version:                          mozart2.VersionV1,
		Action:                           mozart2.AccountBalanceAction,
		UseMozartMappedInternalErrorCode: mozart2.UseMozartMappedInternalErrorCode,
		Timeout:                          mozart2.Timeout,
		ConnectTimeout:                   mozart2.ConnectTimeout,
	}
}

// ValidateRequest validates that all required fields are present in the request
func ValidateRequest(request models.MozartBalanceFetchRequest) errors.IError {
	err := validation.ValidateStruct(&request,
		validation.Field(&request.SourceAccount, validation.Required),
	)
	if err != nil {
		return errtype.BadRequest.New(errcode.ValidationFailure).Wrap(err)
	}

	srcAccount := request.SourceAccount
	err = validation.ValidateStruct(&srcAccount,
		validation.Field(&srcAccount.AccountNumber, validation.Required),
		validation.Field(&srcAccount.Credentials, validation.Required),
	)
	if err != nil {
		return errtype.BadRequest.New(errcode.ValidationFailure).Wrap(err)
	}

	creds := srcAccount.Credentials
	// Add bank-specific credential validation here
	err = validation.ValidateStruct(&creds,
		validation.Field(&creds.FieldName, validation.Required),
		// Add other required fields based on bank requirements
	)
	if err != nil {
		return errtype.BadRequest.New(errcode.ValidationFailure).Wrap(err)
	}

	return nil
}
